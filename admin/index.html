// ============ ADMIN LOGIN - CONNECTED TO SERVER ============ function handleAdminLogin(event) { event.preventDefault(); const email = document.getElementById('email').value; const password = document.getElementById('password').value; const errorDiv = document.getElementById('loginError'); const submitBtn = document.querySelector('.btn-primary'); // Show loading submitBtn.disabled = true; submitBtn.innerHTML = ' Logging in...'; errorDiv.style.display = 'none'; // Send to real server fetch('/api/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ identifier: email, password: password }) }) .then(res => res.json()) .then(data => { if (data.success) { // Check if admin if (data.user.role === 'admin') { localStorage.setItem('apexAdminLoggedIn', 'true'); localStorage.setItem('apexAdminUser', data.user.email); localStorage.setItem('apexAdminToken', data.token); window.location.href = 'dashboard.html'; } else { errorDiv.style.display = 'block'; errorDiv.innerHTML = ' Access denied. Admin only.'; submitBtn.disabled = false; submitBtn.innerHTML = ' Login to Dashboard'; } } else { errorDiv.style.display = 'block'; errorDiv.innerHTML = ' ' + data.message; submitBtn.disabled = false; submitBtn.innerHTML = ' Login to Dashboard'; } }) .catch(err => { errorDiv.style.display = 'block'; errorDiv.innerHTML = ' Server error. Is the server running?'; submitBtn.disabled = false; submitBtn.innerHTML = ' Login to Dashboard'; }); } // Add default admin account on first run function createDefaultAdmin() { fetch('/api/auth/register', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ fullName: 'Admin User', mobile: '9999999999', email: 'admin@apexhub.in', password: 'admin123', role: 'admin' }) }) .then(res => res.json()) .then(data => { if (data.success) { console.log('✅ Default admin created'); } }) .catch(() => {}); } // Run on page load createDefaultAdmin();
